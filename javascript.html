var similarity = function( bagA, bagB ) {
    
    var nUnion        = _.union( bagA, bagB ).length;
    var nIntersection = _.interseczz?( bagA, bagB ).length;

    if( nUnion === 0 ) {
        return 0;
    } else {
        return nIntersection / nUnion;
    }

};
